\newcolumntype{Y}{>{\centering\arraybackslash}X}

\newenvironment{layouttable}[2]{
    \let\oldtabcolsep\tabcolsep
    \setlength{\tabcolsep}{0em}
    \tabularx{#1}{#2}
}{
    \endtabularx
    \setlength{\tabcolsep}{\oldtabcolsep}
}

\newenvironment{flexiblecolumns}[1][1]{
    \let\oldtabcolsep\tabcolsep
    \setlength{\tabcolsep}{0em}
    \tabularx{\textwidth}{lY*{#1}r}
}{
    \endtabularx
    \setlength{\tabcolsep}{\oldtabcolsep}
}

\newcommand{\underhint}[1]{

}

\newcommand{\prefill}[1]{
